    \node[const, above = of theta]					(alpha)		{$\alpha$};
    \node[latent, above = of f]		(a)		{$ a $};
    \node[cond, left = of f]		(e)		{$ e_0^{m} $};
    \node[cond, above = of e]		(m)		{$ m $};
    
    % Connect nodes
    \edge{e,a}{f};
    \edge{m}{a};
    \edge{theta}{f};
    \edge{alpha}{theta};
